340 void updateParametersFromStream(Stream* inStream, int timeout){
341  if(!inStream->available())return; //if there is no data available, dont even start timeouts
342  //read data from stream
343  char buf[PAR_SETTINGS_BUFFER_SIZE];
344  int bufPos=0;
345  int inByte;
346  long lastCharMillis=millis();
347  //even if the buffer is full, we keep on reading to flush the remaining data.
348  while((millis()-lastCharMillis)<timeout){
349  inByte=inStream->read();
350  if(inByte>0){
351  lastCharMillis=millis();
352  if(bufPos<PAR_SETTINGS_BUFFER_SIZE-1){
353  buf[bufPos]=inByte;
354  bufPos++;
355  }
356  }
357  }
358 
359  buf[bufPos]=0;
360 
361  //let all the parameter setting instances have a look at the data
362  if(bufPos>0){
363  for(int i=0;i<PAR_SETTINGS_CUR_INSTANCE_NUMBER;i++){
364  PAR_SETTINGS_INSTANCES[i]->parseSerialData(buf);
365  }
366  }
367 };
