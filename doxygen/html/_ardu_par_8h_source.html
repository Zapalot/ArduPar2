369 void updateParametersFromStream(Stream* inStream, int timeout){
370  if(!inStream->available())return; //if there is no data available, dont even start timeouts
371  //read data from stream
372  char buf[PAR_SETTINGS_BUFFER_SIZE];
373  int bufPos=0;
374  int inByte;
375  long lastCharMillis=millis();
376  //even if the buffer is full, we keep on reading to flush the remaining data.
377  while((millis()-lastCharMillis)<timeout){
378  inByte=inStream->read();
379  if(inByte>0){
380  lastCharMillis=millis();
381  if(bufPos<PAR_SETTINGS_BUFFER_SIZE-1){
382  buf[bufPos]=inByte;
383  bufPos++;
384  }
385  }
386  }
387 
388  buf[bufPos]=0;
389 
390  //let all the parameter setting instances have a look at the data
391  if(bufPos>0){
392  for(int i=0;i<PAR_SETTINGS_CUR_INSTANCE_NUMBER;i++){
393  PAR_SETTINGS_INSTANCES[i]->parseSerialData(buf);
394  }
395  }
